   ```
   ├── cassava

   | | ├── cbb
   | | ├── cbsd
   | | ├── cgm
   | | ├── cmd
   | | ├── healthy
   | ├── test
   | | ├── 0
   | ├── extraimages
   | | ├── 0
   ├── dataloaders
   ├── networks
   ├── utils
   ├── config.py
   ├── main.py
   └── README.md
   ```
